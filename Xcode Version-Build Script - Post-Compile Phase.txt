echo "============================================="
echo "Xcode Version-Build Script - Post-Compile Phase"

useGitCommitCount=$(/usr/libexec/PlistBuddy -c "Print useGitCommitCount" "${SRCROOT}/rootInfo.plist")

echo "Project root: ${SRCROOT}"
echo "Info.plist: ${INFOPLIST_FILE}"
echo "Configuration: ${CONFIGURATION}"
echo "useGitCommitCount: $useGitCommitCount"
echo .

# Only increase the build number on release configuration
if [ ${CONFIGURATION} == "Release" ]; then

bundleShortVersion=`/usr/libexec/PlistBuddy -c "Print CFBundleShortVersionString" "${INFOPLIST_FILE}"`
numberOfOccurrences=$(grep -o "\." <<< "$bundleShortVersion" | wc -l)

if [ $numberOfOccurrences == "1" ]; then

bundleShortVersion=`/usr/libexec/PlistBuddy -c "Print CFBundleShortVersionString" "${INFOPLIST_FILE}" | /usr/bin/perl -pe 's/(\d+\.)(\d+)/$1.($2+1)/eg'`

else

bundleShortVersion=`/usr/libexec/PlistBuddy -c "Print CFBundleShortVersionString" "${INFOPLIST_FILE}" | /usr/bin/perl -pe 's/(\d+\.)(\d+\.)(\d+)/$1.$2.($3+1)/eg'`

fi

echo "Increment version number to $bundleShortVersion"
echo .
/usr/libexec/PListBuddy -c "Set :CFBundleShortVersionString $bundleShortVersion" "${INFOPLIST_FILE}"

#Reset CFBundleVersion if NOT using git commit count
if [ $useGitCommitCount == "NO" ]; then

bundleVersion=`/usr/libexec/PlistBuddy -c "Print CFBundleVersion" "${INFOPLIST_FILE}" | /usr/bin/perl -pe 's/(\d+)/($1-$1)/eg'`
echo "Reset build number to $bundleVersion"
echo .
/usr/libexec/PListBuddy -c "Set :CFBundleVersion $bundleVersion" "${INFOPLIST_FILE}"

fi

fi

echo "Sync all Info.plist files in the project"

bundleShortVersion=$(/usr/libexec/PlistBuddy -c "Print CFBundleShortVersionString" "${INFOPLIST_FILE}")
bundleVersion=$(/usr/libexec/PlistBuddy -c "Print CFBundleVersion" "${INFOPLIST_FILE}")

find "${SRCROOT}" -name 'Info.plist' | while read productInfoPlistFile; do
echo "$productInfoPlistFile $bundleShortVersion ($bundleVersion)"
/usr/libexec/PListBuddy -c "Set :CFBundleShortVersionString $bundleShortVersion" "$productInfoPlistFile"
/usr/libexec/PListBuddy -c "Set :CFBundleVersion $bundleVersion" "$productInfoPlistFile"
done

echo .
echo "Reomve useGitCommitCount in rootInfo.plist and delete the file"
/usr/libexec/PListBuddy -c "Delete :useGitCommitCount" "${SRCROOT}/rootInfo.plist"
rm "${SRCROOT}/rootInfo.plist"
echo "============================================="
