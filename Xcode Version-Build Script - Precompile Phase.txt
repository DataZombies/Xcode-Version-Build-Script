echo "============================================="
echo "Xcode Version-Build Script - Precompile Phase"

useGitCommitCount=$"YES" # "YES"|"NO"

echo "Project root: ${SRCROOT}"
echo "Info.plist: ${INFOPLIST_FILE}"
echo "useGitCommitCount: $useGitCommitCount"
echo .
echo "Save useGitCommitCount to project root infoPlistFile"
/usr/libexec/PListBuddy -c "Add :useGitCommitCount string $useGitCommitCount" "${SRCROOT}/rootInfo.plist"
echo .

# Use this repository's git commit count for the CFBundleVersion
if [ $useGitCommitCount == "YES" ]; then

bundleVersion=$(git rev-list HEAD | wc -l | tr -d ' ')
echo "Using git commit count for CFBundleVersion: $bundleVersion"
/usr/libexec/PlistBuddy -c "Set :CFBundleVersion $bundleVersion" "${INFOPLIST_FILE}"

else

# Increment the value in bundleVersion for the CFBundleVersion
bundleVersion=`/usr/libexec/PlistBuddy -c "Print CFBundleVersion" "${INFOPLIST_FILE}" | /usr/bin/perl -pe 's/(\d+)/($1+1)/eg'`
echo "Increment CFBundleVersion to $bundleVersion"
/usr/libexec/PListBuddy -c "Set :CFBundleVersion $bundleVersion" "${INFOPLIST_FILE}"

fi

echo .
echo "Sync all Info.plist files in the project"

bundleShortVersion=$(/usr/libexec/PlistBuddy -c "Print CFBundleShortVersionString" "${INFOPLIST_FILE}")
bundleVersion=$(/usr/libexec/PlistBuddy -c "Print CFBundleVersion" "${INFOPLIST_FILE}")

find "${SRCROOT}" -name 'Info.plist' | while read productInfoPlistFile; do
echo "$productInfoPlistFile $bundleShortVersion ($bundleVersion)"
/usr/libexec/PListBuddy -c "Set :CFBundleShortVersionString $bundleShortVersion" "$productInfoPlistFile"
/usr/libexec/PListBuddy -c "Set :CFBundleVersion $bundleVersion" "$productInfoPlistFile"
done
echo "============================================="
